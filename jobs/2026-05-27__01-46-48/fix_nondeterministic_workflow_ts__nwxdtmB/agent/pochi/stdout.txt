export async function pickDiscount(): Promise<number> {
  const options = [0, 5, 10, 15, 20];
  const idx = Math.floor(Math.random() * options.length);

### `src/workflows.ts` — fully refactored
```ts
import { proxyActivities } from '@temporalio/workflow';
import type * as activities from './activities';

const { pickDiscount } = proxyActivities<typeof activities>({
  startToCloseTimeout: '10 seconds',
});

export async function DiscountWorkflow(userId: string): Promise<DiscountResult> {
  const discount = await pickDiscount();   // delegated to Activity
  const decidedAt = Date.now();            // SDK-patched: deterministic in Workflow context
  return { userId, discount, decidedAt };
}
```
- `Math.random` and `new Date(` have been **completely removed** from the Workflow body.
- Random selection is now performed in the `pickDiscount` Activity, invoked via `proxyActivities` (producing an `ActivityTaskScheduled` event in the history).
- `Date.now()` is used for the timestamp — the Temporal SDK patches this inside Workflow code to return the frozen, deterministic Workflow Task time, making it replay-safe.
